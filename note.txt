from __future__ import annotations

from collections import OrderedDict

import torch
import lightning.pytorch as pl
from lightning.pytorch.callbacks import Callback


class TrainableOnlyEMACallback(Callback):
    def __init__(self, decay: float = 0.999) -> None:
        super().__init__()
        assert 0.0 < decay < 1.0
        self.decay = decay
        self.ema_state: OrderedDict[str, torch.Tensor] = OrderedDict()
        self.backup_state: OrderedDict[str, torch.Tensor] = OrderedDict()

    def _named_trainable_parameters(
        self, pl_module: pl.LightningModule
    ) -> list[tuple[str, torch.nn.Parameter]]:
        return [
            (name, param)
            for name, param in pl_module.named_parameters()
            if param.requires_grad
        ]

    @torch.no_grad()
    def on_fit_start(
        self, trainer: pl.Trainer, pl_module: pl.LightningModule
    ) -> None:
        self.ema_state = OrderedDict(
            (name, param.detach().clone())
            for name, param in self._named_trainable_parameters(pl_module)
        )

    @torch.no_grad()
    def on_train_batch_end(
        self, trainer: pl.Trainer, pl_module: pl.LightningModule, outputs, batch, batch_idx: int
    ) -> None:
        for name, param in self._named_trainable_parameters(pl_module):
            self.ema_state[name].mul_(self.decay).add_(param.detach(), alpha=1.0 - self.decay)

    @torch.no_grad()
    def _swap_in_ema_weights(self, pl_module: pl.LightningModule) -> None:
        self.backup_state = OrderedDict()
        param_dict = dict(pl_module.named_parameters())

        for name, ema_param in self.ema_state.items():
            param = param_dict[name]
            self.backup_state[name] = param.detach().clone()
            param.data.copy_(ema_param.data)

    @torch.no_grad()
    def _restore_original_weights(self, pl_module: pl.LightningModule) -> None:
        param_dict = dict(pl_module.named_parameters())

        for name, orig_param in self.backup_state.items():
            param_dict[name].data.copy_(orig_param.data)

        self.backup_state = OrderedDict()

    def on_validation_epoch_start(
        self, trainer: pl.Trainer, pl_module: pl.LightningModule
    ) -> None:
        if self.ema_state:
            self._swap_in_ema_weights(pl_module)

    def on_validation_epoch_end(
        self, trainer: pl.Trainer, pl_module: pl.LightningModule
    ) -> None:
        if self.backup_state:
            self._restore_original_weights(pl_module)

    def on_test_epoch_start(
        self, trainer: pl.Trainer, pl_module: pl.LightningModule
    ) -> None:
        if self.ema_state:
            self._swap_in_ema_weights(pl_module)

    def on_test_epoch_end(
        self, trainer: pl.Trainer, pl_module: pl.LightningModule
    ) -> None:
        if self.backup_state:
            self._restore_original_weights(pl_module)

    def on_save_checkpoint(
        self, trainer: pl.Trainer, pl_module: pl.LightningModule, checkpoint: dict
    ) -> dict:
        checkpoint["trainable_only_ema_state"] = self.ema_state
        checkpoint["trainable_only_ema_decay"] = self.decay
        return checkpoint

    def on_load_checkpoint(
        self, trainer: pl.Trainer, pl_module: pl.LightningModule, checkpoint: dict
    ) -> None:
        loaded = checkpoint.get("trainable_only_ema_state")
        if loaded is not None:
            self.ema_state = loaded
